from abc import ABC, abstractmethod

class Animal(ABC):

    def __init__(self, health, dmg, position):
        self.health = health
        self.dmg = dmg
        self.position = position
        self.alive = True
        self.level = 1 # above 5: level 3, above 3: level 2, less: level 1

    def onHit(self, dmgAmt):
        self.dmg -= dmgAmt
        if self.dmg <= 0:
            self.alive = False

    def onCombine(self):
        self.level += 1     

    def updatePosition(self, pos):
        self.position = pos      

    @abstractmethod
    def onfaint(self):
        pass
